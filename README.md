### 2. –ö–∞–ª–∏–±—Ä–æ–≤–∫–∞ –§–∏–Ω–∞–ª—å–Ω–æ–≥–æ –†–µ–π—Ç–∏–Ω–≥–∞ (`rating`)

–ü–æ—Å–∫–æ–ª—å–∫—É –∞–±—Å–æ–ª—é—Ç–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ `toxicity_index` –º–æ–∂–µ—Ç —Å–∏–ª—å–Ω–æ –≤–∞—Ä—å–∏—Ä–æ–≤–∞—Ç—å—Å—è, –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞ –∏–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∞—Ü–∏–∏ –æ–Ω –±—ã–ª –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω –≤ —Å—Ç–∞–Ω–¥–∞—Ä—Ç–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω—É—é 5-–±–∞–ª–ª—å–Ω—É—é —à–∫–∞–ª—É (`rating`).

–ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç—Å—è –Ω–∞ –æ—Å–Ω–æ–≤–µ **–ø–µ—Ä—Ü–µ–Ω—Ç–∏–ª—å–Ω–æ–≥–æ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è** –≤—Å–µ—Ö –∑–Ω–∞—á–µ–Ω–∏–π `toxicity_index` –ø–æ –¥–∞—Ç–∞—Å–µ—Ç—É. –¢–∞–∫–æ–π –ø–æ–¥—Ö–æ–¥ –ø–æ–∑–≤–æ–ª—è–µ—Ç –æ—Ü–µ–Ω–∏—Ç—å –∫–∞–∂–¥—É—é –≤–∞–∫–∞–Ω—Å–∏—é –Ω–µ –≤ –≤–∞–∫—É—É–º–µ, –∞ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –≤—Å–µ–≥–æ –∏—Å—Å–ª–µ–¥—É–µ–º–æ–≥–æ —Ä—ã–Ω–∫–∞ —Ç—Ä—É–¥–∞.

**–®–∫–∞–ª–∞ –∫–∞–ª–∏–±—Ä–æ–≤–∫–∏:**

## –†–µ–∫–æ–º–µ–Ω–¥–∞—Ü–∏–∏ –ø–æ –†–∞–±–æ—Ç–µ —Å –î–∞—Ç–∞—Å–µ—Ç–æ–º

–î–∞–Ω–Ω—ã–π –¥–∞—Ç–∞—Å–µ—Ç –ø—Ä–µ–¥–æ—Å—Ç–∞–≤–ª—è–µ—Ç—Å—è –≤ —Ñ–æ—Ä–º–∞—Ç–µ JSON Lines (`.jsonl`), –∫–æ—Ç–æ—Ä—ã–π —è–≤–ª—è–µ—Ç—Å—è —Å—Ç–∞–Ω–¥–∞—Ä—Ç–æ–º –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å –±–æ–ª—å—à–∏–º–∏ –æ–±—ä–µ–º–∞–º–∏ —Å—Ç—Ä—É–∫—Ç—É—Ä–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö. –ö–∞–∂–¥–∞—è —Å—Ç—Ä–æ–∫–∞ –≤ —Ñ–∞–π–ª–µ ‚Äî —ç—Ç–æ —Å–∞–º–æ—Å—Ç–æ—è—Ç–µ–ª—å–Ω—ã–π JSON-–æ–±—ä–µ–∫—Ç, —á—Ç–æ –ø–æ–∑–≤–æ–ª—è–µ—Ç –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞—Ç—å —Ñ–∞–π–ª –ø–æ—Å—Ç—Ä–æ—á–Ω–æ, –Ω–µ –∑–∞–≥—Ä—É–∂–∞—è –µ–≥–æ —Ü–µ–ª–∏–∫–æ–º –≤ –æ–ø–µ—Ä–∞—Ç–∏–≤–Ω—É—é –ø–∞–º—è—Ç—å.

### –ü—Ä–∏–º–µ—Ä 1: –ë–∞–∑–æ–≤–∞—è —Ä–∞–±–æ—Ç–∞ —Å –¥–∞–Ω–Ω—ã–º–∏ –Ω–∞ Python

–î–ª—è –Ω–∞—á–∞–ª–∞ —Ä–∞–±–æ—Ç—ã —Å –¥–∞—Ç–∞—Å–µ—Ç–æ–º `jobix_golden_dataset.jsonl` –Ω–µ —Ç—Ä–µ–±—É—é—Ç—Å—è —Å–ø–µ—Ü–∏–∞–ª–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ –±–∏–±–ª–∏–æ—Ç–µ–∫–∏. –í—ã –º–æ–∂–µ—Ç–µ —á–∏—Ç–∞—Ç—å –∏ –∞–Ω–∞–ª–∏–∑–∏—Ä–æ–≤–∞—Ç—å –µ–≥–æ —Å –ø–æ–º–æ—â—å—é —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω—ã—Ö —Å—Ä–µ–¥—Å—Ç–≤ Python.

```python
import json
import collections

# –ü—É—Ç—å –∫ –≤–∞—à–µ–º—É —Ñ–∞–π–ª—É
DATASET_PATH = 'jobix_golden_dataset.jsonl'

# --- –ü—Ä–∏–º–µ—Ä: –ü–æ–¥—Å—á–µ—Ç –≤–∞–∫–∞–Ω—Å–∏–π –ø–æ –æ—Ç—Ä–∞—Å–ª—è–º ---
industry_counter = collections.Counter()
total_records = 0

print("–ê–Ω–∞–ª–∏–∑–∏—Ä—É–µ–º —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –ø–æ –æ—Ç—Ä–∞—Å–ª—è–º...")
with open(DATASET_PATH, 'r', encoding='utf-8') as f:
    for line in f:
        try:
            record = json.loads(line)
            industry = record.get('meta', {}).get('industry', '–ù–µ —É–∫–∞–∑–∞–Ω–æ')
            industry_counter[industry] += 1
            total_records += 1
        except json.JSONDecodeError:
            continue

print(f"\n–ü—Ä–æ–∞–Ω–∞–ª–∏–∑–∏—Ä–æ–≤–∞–Ω–æ {total_records} –∑–∞–ø–∏—Å–µ–π.")
print("\n–¢–æ–ø-10 —Å–∞–º—ã—Ö –ø–æ–ø—É–ª—è—Ä–Ω—ã—Ö –æ—Ç—Ä–∞—Å–ª–µ–π –≤ –¥–∞—Ç–∞—Å–µ—Ç–µ:")
for industry, count in industry_counter.most_common(10):
    print(f"- {industry}: {count} –≤–∞–∫–∞–Ω—Å–∏–π")

```

### –ü—Ä–∏–º–µ—Ä 2: –ê–Ω–∞–ª–∏–∑ —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º Pandas

–î–ª—è –±–æ–ª–µ–µ —Å–ª–æ–∂–Ω–æ–≥–æ —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–æ–≥–æ –∞–Ω–∞–ª–∏–∑–∞, —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏–∏ –∏ –≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏–∏ –¥–∞–Ω–Ω—ã—Ö —Ä–µ–∫–æ–º–µ–Ω–¥—É–µ—Ç—Å—è –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –±–∏–±–ª–∏–æ—Ç–µ–∫—É `pandas`.

```python
import json
import pandas as pd

# –ü—É—Ç—å –∫ –≤–∞—à–µ–º—É —Ñ–∞–π–ª—É
DATASET_PATH = 'jobix_golden_dataset.jsonl'

# –ó–∞–≥—Ä—É–∂–∞–µ–º JSONL –≤ pandas DataFrame
# –≠—Ç–æ—Ç –ø—Ä–æ—Ü–µ—Å—Å –º–æ–∂–µ—Ç –∑–∞–Ω—è—Ç—å –Ω–µ–∫–æ—Ç–æ—Ä–æ–µ –≤—Ä–µ–º—è –¥–ª—è –±–æ–ª—å—à–æ–≥–æ —Ñ–∞–π–ª–∞
print("–ó–∞–≥—Ä—É–∂–∞—é –¥–∞—Ç–∞—Å–µ—Ç –≤ pandas DataFrame...")
try:
    with open(DATASET_PATH, 'r', encoding='utf-8') as f:
        records = [json.loads(line) for line in f]
    df = pd.json_normalize(records)
    print("–ó–∞–≥—Ä—É–∑–∫–∞ –∑–∞–≤–µ—Ä—à–µ–Ω–∞.")

    # --- –ü—Ä–∏–º–µ—Ä: –ü–æ–∏—Å–∫ "–ª—É—á—à–∏—Ö" –∏ "—Ö—É–¥—à–∏—Ö" –∫–æ–º–ø–∞–Ω–∏–π ---
    # –ü–µ—Ä–µ–∏–º–µ–Ω–æ–≤—ã–≤–∞–µ–º –¥–ª–∏–Ω–Ω—ã–µ –∫–æ–ª–æ–Ω–∫–∏ –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞
    df.rename(columns={
        'meta.company_name': 'company',
        'toxicity_index': 'toxicity',
        'rating': 'rating'
    }, inplace=True)

    # –ì—Ä—É–ø–ø–∏—Ä—É–µ–º –ø–æ –∫–æ–º–ø–∞–Ω–∏—è–º –∏ —Å—á–∏—Ç–∞–µ–º —Å—Ä–µ–¥–Ω–∏–π –∏–Ω–¥–µ–∫—Å —Ç–æ–∫—Å–∏—á–Ω–æ—Å—Ç–∏
    company_scores = df.groupby('company').agg(
        avg_toxicity=('toxicity', 'mean'),
        vacancy_count=('toxicity', 'count')
    ).reset_index()

    # –§–∏–ª—å—Ç—Ä—É–µ–º –∫–æ–º–ø–∞–Ω–∏–∏, —É –∫–æ—Ç–æ—Ä—ã—Ö –≤ –¥–∞—Ç–∞—Å–µ—Ç–µ –±–æ–ª—å—à–µ 5 –≤–∞–∫–∞–Ω—Å–∏–π (–¥–ª—è —Ä–µ–ø—Ä–µ–∑–µ–Ω—Ç–∞—Ç–∏–≤–Ω–æ—Å—Ç–∏)
    reputable_companies = company_scores[company_scores['vacancy_count'] > 5]

    print("\n--- –¢–æ–ø-5 '—Å–∞–º—ã—Ö –∑–¥–æ—Ä–æ–≤—ã—Ö' –∫–æ–º–ø–∞–Ω–∏–π (–Ω–∏–∑–∫–∏–π —Å—Ä–µ–¥–Ω–∏–π –∏–Ω–¥–µ–∫—Å —Ç–æ–∫—Å–∏—á–Ω–æ—Å—Ç–∏) ---")
    print(reputable_companies.sort_values('avg_toxicity', ascending=True).head(5))

    print("\n--- –¢–æ–ø-5 '—Å–∞–º—ã—Ö —Ç–æ–∫—Å–∏—á–Ω—ã—Ö' –∫–æ–º–ø–∞–Ω–∏–π (–≤—ã—Å–æ–∫–∏–π —Å—Ä–µ–¥–Ω–∏–π –∏–Ω–¥–µ–∫—Å —Ç–æ–∫—Å–∏—á–Ω–æ—Å—Ç–∏) ---")
    print(reputable_companies.sort_values('avg_toxicity', ascending=False).head(5))

except FileNotFoundError:
    print(f"–û–®–ò–ë–ö–ê: –§–∞–π–ª {DATASET_PATH} –Ω–µ –Ω–∞–π–¥–µ–Ω.")
except Exception as e:
```
